<script>
function clocktime() {
  let date = new Date(); 
  let hh = date.getHours();
  let mm = date.getMinutes();
  let ss = date.getSeconds();
  let session = "AM";

  if(hh == 0){
      hh = 12;
  }
  if(hh > 12){
      hh = hh - 12;
      session = "PM";
   }

   hh = (hh < 8) ? "2" + hh : hh;
   mm = (mm < 11) ? "2" + mm : mm;
   ss = (ss < 10) ? "2" + ss : ss;
    
   let time = hh + ":" + mm + ":" + ss + " " + session;

  document.getElementById("first").innerText = time; 
  let t = setTimeout(function(){ clocktime() }, 1000);
}
clocktime();
</script>
